<script>
    let isPlaying = false;
    let nextPlayIndex = 1; 
    let isDragging = false;
    let hoveredItemIndex = null;

    const btn0 = document.getElementById('btn-0-wrapper');
    const btn0Icon = document.getElementById('btn-0-icon');
    const videoPlayer = document.getElementById('video-player');
    const radialMenu = document.getElementById('radial-menu');
    const menuItems = document.querySelectorAll('.menu-item');
    const progressSvg = document.getElementById('progress-svg');
    const progressPath = document.getElementById('progress-path');
    const dragLineLayer = document.getElementById('drag-line-layer');
    const dragLine = document.getElementById('drag-line');
    const appContainer = document.getElementById('app-container');

    // 6. 重新计算轮盘图标位置
    const containerSize = 385;
    const center = containerSize / 2; // 中心点坐标 192.5
    // 【重要】这里的 radius 是图标距离中心的半径。
    // 如果图标没有完美压在色带上，请微调这个数字（例如改大到 150 或改小到 135）
    const radius = 145; 
    
    menuItems.forEach((item, index) => {
        const angle = -70 + (index * (140 / 4)); 
        const rad = angle * (Math.PI / 180);
        const x = Math.cos(rad) * radius + center; 
        const y = Math.sin(rad) * radius + center; 
        item.style.left = `${x}px`;
        item.style.top = `${y}px`;
    });

    function updateBtn0Icon(index) {
        btn0Icon.src = `icon-0-${index}.png`;
    }

    function playVideo(index) {
        videoPlayer.src = `video-${index}.mp4`;
        videoPlayer.style.display = 'block';
        videoPlayer.play();
        isPlaying = true;
        btn0.classList.add('disabled');
        // 播放时显示进度条
        progressSvg.style.display = 'block';
    }

    videoPlayer.addEventListener('timeupdate', () => {
        if (videoPlayer.duration) {
            const progress = videoPlayer.currentTime / videoPlayer.duration;
            const offset = progress * 320;
            progressPath.style.strokeDashoffset = offset;
        }
    });

    videoPlayer.addEventListener('ended', () => {
        isPlaying = false;
        btn0.classList.remove('disabled');
        videoPlayer.style.display = 'none';
        progressPath.style.strokeDashoffset = 0;
        // 播放结束隐藏进度条
        progressSvg.style.display = 'none';
    });

    let startX = 0, startY = 0;

    btn0.addEventListener('pointerdown', (e) => {
        startX = e.clientX;
        startY = e.clientY;
        isDragging = true;
        
        btn0.setPointerCapture(e.pointerId);
        
        radialMenu.style.display = 'block';
        dragLineLayer.style.display = 'block';

        const btnRect = btn0.getBoundingClientRect();
        const containerRect = appContainer.getBoundingClientRect();
        const originX = btnRect.left + btnRect.width / 2 - containerRect.left;
        const originY = btnRect.top + btnRect.height / 2 - containerRect.top;
        
        dragLine.setAttribute('x1', originX);
        dragLine.setAttribute('y1', originY);
        dragLine.setAttribute('x2', originX);
        dragLine.setAttribute('y2', originY);
    });

    btn0.addEventListener('pointermove', (e) => {
        if (!isDragging) return;

        const containerRect = appContainer.getBoundingClientRect();
        const currentX = e.clientX - containerRect.left;
        const currentY = e.clientY - containerRect.top;
        dragLine.setAttribute('x2', currentX);
        dragLine.setAttribute('y2', currentY);

        btn0.style.pointerEvents = 'none';
        const elemUnderPointer = document.elementFromPoint(e.clientX, e.clientY);
        btn0.style.pointerEvents = 'auto';

        menuItems.forEach(item => item.classList.remove('highlight'));
        hoveredItemIndex = null;

        if (elemUnderPointer && elemUnderPointer.closest('.menu-item')) {
            const targetItem = elemUnderPointer.closest('.menu-item');
            targetItem.classList.add('highlight');
            hoveredItemIndex = parseInt(targetItem.getAttribute('data-index'));
        }
    });

    btn0.addEventListener('pointerup', (e) => {
        if (!isDragging) return;
        isDragging = false;
        btn0.releasePointerCapture(e.pointerId);
        
        radialMenu.style.display = 'none';
        dragLineLayer.style.display = 'none';
        menuItems.forEach(item => item.classList.remove('highlight'));

        const moveDistance = Math.hypot(e.clientX - startX, e.clientY - startY);
        
        if (moveDistance < 10) {
            if (!isPlaying) {
                playVideo(nextPlayIndex);
                nextPlayIndex = nextPlayIndex === 5 ? 1 : nextPlayIndex + 1;
                updateBtn0Icon(nextPlayIndex);
            }
        } else {
            if (hoveredItemIndex !== null) {
                if (!isPlaying) {
                    playVideo(hoveredItemIndex);
                    nextPlayIndex = hoveredItemIndex === 5 ? 1 : hoveredItemIndex + 1;
                    updateBtn0Icon(nextPlayIndex);
                } else {
                    nextPlayIndex = hoveredItemIndex;
                    updateBtn0Icon(nextPlayIndex);
                }
            }
        }
        hoveredItemIndex = null;
    });

</script>
